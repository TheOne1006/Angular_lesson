过滤器filter: 字符串 {{ ['Ari','Lerner','Likes','To','Eat','Pizza'] | filter:'e' }}

过滤器filter: 对象{属性名:值} {{ [{ 'name': 'Ari', 'City': 'San Francisco', 'favorite food': 'Pizza' },{ 'name': 'Nate', 'City': 'San Francisco', 'favorite food': 'indian food' }] | filter:{'favorite food': 'Pizza'} }}

过滤器filter: 函数 {{ ['Ari','likes','to','travel'] | filter:isCapitalized }}

过滤器json - 将JSON 对象转换成字符串 {{ {'name': 'Ari', 'City': 'SanFrancisco'} | json }}

过滤器limiTo: {{ 'today is Monday!' | limitTo:5 }} {{ ['1','2'] | limitTo:1 }}

lowercase {{ 'Da Xie 转小写' | lowercase }}

orderBy用表达式对指定的数据进行排序，两个参数 第一个必须，第二个可选， {{ [{ 'name': 'Ari', 'status': 'awake' },{ 'name': 'Q', 'status': 'sleeping' },{ 'name': 'Nate', 'status': 'awake' }] | orderBy:'name' }}

首字母大写： {{ 'to first Uplowwer' | lowercase }}
